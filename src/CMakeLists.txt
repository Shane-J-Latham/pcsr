
SET(
  pcsr_SRCS
  pcsr_cgal.cpp
  pcsr_pcl.cpp
  pcsr.cpp
)

ADD_LIBRARY(pcsr MODULE ${pcsr_SRCS})
SET_TARGET_PROPERTIES(pcsr PROPERTIES PREFIX "" )
SET_TARGET_PROPERTIES(pcsr PROPERTIES OUTPUT_NAME _pcsr )
TARGET_LINK_LIBRARIES(
  pcsr
  ${CGAL_LIBRARIES}
  ${PCSR_PCL_LIBRARIES}
  ${BOOST_LIBRARIES}
  ${PCSR_BOOST_PYTHON_LIBRARIES}
  ${PYTHON_LIBRARIES}
  ${MPFR_LIBRARIES}
  ${GMP_LIBRARIES}
)
SET_TARGET_PROPERTIES(pcsr PROPERTIES INSTALL_NAME_DIR ${PCSR_PYTHON_SITE_PACKAGES_INSTALL_DIR} )
INSTALL(TARGETS pcsr DESTINATION ${PCSR_PYTHON_SITE_PACKAGES_INSTALL_DIR})
